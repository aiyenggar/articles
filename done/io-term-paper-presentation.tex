\begin{frame}{Core Tenets of Transaction Cost Economics}
\begin{itemize}
\item<1->{\begin{hypothesis}{ As asset specificity increases, the transaction costs associated with market governance increase}\end{hypothesis}}
\item<2->{\begin{hypothesis}{As asset specificity increases, hybrids and hierarchies become preferred over markets; at high levels of asset specificity, hierarchy becomes the preferred governance form}\end{hypothesis}}
\item<3->{\begin{hypothesis}{When asset specificity is present to a nontrivial degree, uncertainty raises the transaction costs associated with market governance}\end{hypothesis}}
\item<4->{\begin{hypothesis}{When asset specificity is present to a nontrivial degree, increasing uncertainty renders markets preferable to hybrids, and hierarchies preferable to both hybrids and markets}\end{hypothesis}}
\end{itemize}
\cite{David2004}
\end{frame}

\begin{frame}{Core Tenets of Transaction Cost Economics}
\begin{itemize}
\item<1->{\begin{hypothesis}{When both asset specificity and uncertainty are high, hierarchy is the most cost-effective governance mode}\end{hypothesis}}
\item<2->{\begin{hypothesis}{Governance modes that are aligned with transaction characteristics should display performance advantages over other modes; for example, when both asset specificity and uncertainty are high, hierarchy should display performance advantages over markets and hybrids}\end{hypothesis}}
\end{itemize}
\cite{David2004}
\end{frame}
